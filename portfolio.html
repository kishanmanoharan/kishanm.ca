<!-- Python: OpenHardwareMonitor and Arduino Serial --> 
import wmi, serial
from time import sleep

w = wmi.WMI(namespace="root\OpenHardwareMonitor")
temperature_infos = w.Sensor()
ser = serial.Serial('COM4', 115200)
if(ser.isOpen() == False):
    ser.open()

temperature_infos = w.Sensor()

while True:
    comm_out = []
    temperature_infos = w.Sensor()
    for sensor in temperature_infos:
        comm_out.append(sensor.value)
    ser.write(comm_out.encode())
    sleep(1)
